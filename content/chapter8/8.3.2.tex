	\begin{itemize}
		\item Use continue statement inside loops to skip current iteration and continue for the next iteration.
		
		\codeblock{
			for(int i=0; i<10; i++) \{ \\
			\s	if(i \% 2 == 0) \\
			\s \s		continue; \\
			\s	System.out.println(i); \\
			\}
		}
		\bigskip
		\outputblock{
		1 \\
		3 \\
		5 \\
		7 \\
		9
		}
		
	\end{itemize}

\end{flushleft}
