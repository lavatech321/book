		Has-a relationship causes \textbf{reuseability} of the code.
		There are 2 types of Has-A relationship:
		\begin{itemize}
			\item Aggregation (Weaker form of "has-a")
			\item Composition (Strong form of "has-a")
		\end{itemize}
		
		\textbf{Aggregation:}
		\begin{itemize}
			\item Aggregation is a one-way relationship(unidirectional association).
			\item For eg: Bank can have employees but vice versa is not possible.
			\codeblock{
				class Bank \{ \\
				\s String bname; \\
				\s	Bank(String bname) \{ \\
				\s \s	this.bname = bname; \\
				\s	\} \\
				\s	public void display(Customer c) \{ \\
				\s System.out.println(bname); // Output: AXIS\\
				\s System.out.println(name); // Output: Ram\\
				\s	\} \} \\
				class Customer \{ \\
				\s	String name; \\
				\s	Customer(String name) \{ \\
				\s \s	this.name = name; \\
				\s	\} \} \\
				class Branch \{ \\
				\s	public static void main(String[] args) \{ \\
				\s \s		Bank b = new Bank("AXIS"); \\
				\s \s		Customer c = new Customer("Ram"); \\
				\s \s		b.display(c); \\
				\s	\} \}		
			}
			\end{itemize}
		
		\newpage
		\textbf{Composition:}
			\begin{itemize}
				\item In composition, two entities are highly dependent on each other. 
				\item One entity cannot exist without the other. Eg: A car has an engine.
				\codeblock{
					class Car \{ \\
					\s	\textbf{private final Engine engine;}   \\
					\s	String model; \\
					\s	public Car(String model) \{ \\
					\s \s		\textbf{engine  = new Engine("POWERHIGH");} \\
					\s \s		this.model = model; \\
					\s	\}	  \\
					\s	public void show() \{ \\
					\s \s 		System.out.print(model+","); \\
					\s \s	System.out.print(engine.type); \\
					\s	\} \} \\
					class Engine  \{ \\
					\s	String type; \\
					\s	Engine(String type) \{ \\
					\s \s		this.type = type; \\
					\s	\} \} \\
					public class Test \{ \\
					\s	public static void main(String arg[]) \{ \\
					\s \s		\textbf{Car car = new Car("5X");} \\
					\s \s		car.show(); // Output: 5X,POWERHIGH  \\
					\s 	\} 	 \}
				}	
			\end{itemize}
