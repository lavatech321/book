	\begin{itemize}
		\item \textbf{Single inheritance:} 
		\bigskip
		\codeblock{
			class A \{ \} \\
			class B extends A \{ \}
		}
		\bigskip
	
		\item \textbf{Multilevel inheritance:} 
		\codeblock{
			class Grandparent \{ \} \\
			class Parent extends Grandparent \{ \} \\
			class Child extends Parent \{ \}
		}
		\item \textbf{Hierarchical inheritance:}
		\bigskip
		\codeblock{
			class Superclass \{ \} \\
			class Subclass1 extends Superclass \{ \} \\		
			class Subclass2 extends Superclass \{ \}
		}
		
		\item \textbf{Multiple inheritance (through interfaces)}: 
		\begin{itemize}
			\item A java class cant extend more than one class at a time.
			\item Multiple inheritance is not directly supported.
			\item It can be achieved through interfaces. 
			\codeblock{
				interface Interface1 \{ \} \\
				interface Interface2 \{ \} \\
				class MyClass implements Interface1, Interface2 \{ \}	
			}
			\quest{Why Java does not provide support for multiple-inheritance?}{
				There maybe a chance of ambiguity problem. 
			}
			\quest{Why multiple-inheritance is supported by interface?}{
				We shall see this in the interface chapter.
			}
		\end{itemize}

		\item \textbf{Cyclic inheritance:} Not allowed in java. 
		\codeblock{
			class A extends B \{\} \xmark \\
			class B extends A \{\} \xmark \\
			class A extends A \{\} \xmark
		}
	\end{itemize}
	
\end{flushleft}
